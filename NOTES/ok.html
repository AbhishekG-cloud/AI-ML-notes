# -----------------------------
# 5. Spatial Autocorrelation (Moran's I)
# -----------------------------
gdf = gpd.GeoDataFrame(df, geometry=gpd.points_from_xy(df.Longitude, df.Latitude))
w = KNN.from_dataframe(gdf, k=5)
w.transform = 'r'
moran = Moran(gdf['Malaria_Cases'], w)
print(f"Moran's I: {moran.I:.4f}")
print(f"p-value: {moran.p_sim:.4f}")

# 6. Hotspot Analysis (Getis-Ord Gi*)
# -----------------------------
from libpysal.weights import KNN
w = KNN.from_dataframe(gdf, k=5)  # Adjust k if needed
w.transform = 'r'  # Row-standardized weights

g_local = g_local = G_Local(gdf['Malaria_Cases'], w, permutations=99, n_jobs=1)

gdf['GiZ'] = g_local.Zs
gdf.plot(column='GiZ', cmap='coolwarm', legend=True, figsize=(10, 6))
plt.title("Getis-Ord Gi* Z-scores (Hotspot Map)")
